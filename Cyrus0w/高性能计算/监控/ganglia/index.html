    1.2 部署ganglia¶

    说明：部署略，ganglia版本gmond >= v3.4.x ，gmetad v3.4.x ，Ganglia web-2 frontend v3.5.x ，GD v2.x , PHP v5.1 or higher （with: postgres and GD support）同时注意server端还需要先安装以下几个基础包：

    yum install httpd php php-gd php-pgsql rrdtool perl-rrdtool rrdtool-devel rrdtool-python gd python-devel

    1.3 部署postgresql数据库¶

    # 1.在ganglia server端部署数据库
yum install postgresql-server postgresql-devel postgresql

# 2.初始化数据库
postgresql-setup initdb
systemctl start postgresql

# 3.设置密码
su - postgres
psql
\password
\q
exit

# 4.修改配置文件以便登录数据库时必须使用密码
vim /var/lib/pgsql/data/pg_hba.conf

    如图(改为md5):

    # 5.重启
systemctl restart postgresql

# 6.安装psycopg2（用来python调用postgresql数据库）
pip install psycopg2

# 7.下载插件文件并解压，获取数据库表文件
wget http://ftp.surfsara.nl/pub/outgoing/jobmonarch/1.1.2/ganglia_jobmonarch-1.1.2.tar.gz
tar xf ganglia_jobmonarch-1.1.2.tar.gz

# 8.创建jobarchive数据库
su - postgres
createdb jobarchive
cd ganglia_jobmonarch-1.1.2
psql -f jobarchived/job_dbase.sql jobarchive

# 9.查看数据库
\l

    1.4 部署配置jobarchived¶

    # 1.在ganglia服务端下载插件文件并解压(上一步数据库已经下载过)
wget http://ftp.surfsara.nl/pub/outgoing/jobmonarch/1.1.2/ganglia_jobmonarch-1.1.2.tar.gz
tar xf ganglia_jobmonarch-1.1.2.tar.gz

# 2.部署jobarchived
cd ganglia_jobmonarch-1.1.2
cp jobarchived/jobarchived.conf /etc/jobarchived.conf
cp jobarchived/jobarchived.py /usr/local/sbin/jobarchived.py

# 3. 编辑配置文件（绿色说明处根据实际情况修改）
vim /etc/jobarchived.conf

    [DEFAULT]
# Wether or not to run as a daemon in background
#
DAEMONIZE           : 0

# Specify debugging level here (only when _not_ DAEMONIZE)
#
# 11 = XML: metrics
# 10 = XML: host, cluster, grid, ganglia
# 9  = RRD activity, gmetad config parsing
# 8  = RRD file activity
# 6  = SQL
# 1  = daemon threading
# 0  = errors
#
# default: 0
#
DEBUG_LEVEL         : 1

# Enable logging to syslog?
#
USE_SYSLOG          : 1

# What level msg'es should be logged to syslog?
#
# usually: lvl 0 (errors)
#
SYSLOG_LEVEL            : 0

# Which facility to use in syslog
#
# Known:
#       KERN, USER, MAIL, DAEMON, AUTH, LPR,
#       NEWS, UUCP, CRON and LOCAL0 through LOCAL7
#
SYSLOG_FACILITY         : DAEMON

# Where is the gmetad.conf located
#
GMETAD_CONF         : /etc/ganglia/gmetad.conf         `gmetad配置文件路径，一般默认在此路径下` 

# Where to grab XML data from
# Usually: local gmetad (port 8651)
#
# Syntax: <hostname>:<port>
#
ARCHIVE_XMLSOURCE       : localhost:8651            `ganglia服务端地址和端口，因为在本机中，所以这里直接写localhost:8651`

# List of data_source names to archive for
#
# Syntax: [ "<clustername>", "<clustername>" ]
#
ARCHIVE_DATASOURCES     : [ "OGSAAS" ]              `ganglia中设置的集群名称`

# Amount of hours to store in one single archived rrd
#
# If you would like less files you can set this bigger
# but could degrade performance
#
# For now 12 hours seems to work: 2 periods per day
#
ARCHIVE_HOURS_PER_RRD       : 12

# Which metrics to exclude from archiving
# NOTE: This can be a regexp or a string
#
ARCHIVE_EXCLUDE_METRICS     : ".*Temp.*", ".*RPM.*", ".*Version.*", ".*Tag$", "boottime", "gexec", "os.*", "machine_type"

# Where to store the archived rrd's
# 
#ARCHIVE_PATH           : __JOBARCHIVE_RRDS__
ARCHIVE_PATH            : /usr/local/jobmonarch         `自定义rrd数据库数据存储目录，后面插件的web页面要调用`

# Archive's SQL dbase to use
#
# Syntax: <hostname>/<database>
#
JOB_SQL_DBASE           : localhost/jobarchive          `数据库地址和数据库名称`
JOB_SQL_USER                    : postgres              `数据库用户名`

JOB_SQL_PASSWORD        : 1qaz2WSX                      `数据库密码`

# Timeout for jobs in archive
#
# Assume job has already finished while jobarchived was not running
# after this amount of hours: the it will be finished anyway in the database
#
JOB_TIMEOUT         : 168

# Location of rrdtool binary
#
RRDTOOL             : /usr/bin/rrdtool

    # 4.启动作业归档插件
jobarchived.py

# 5.检查插件是否运行在后台
ps aux|grep jobarchived

    1.5 设置ganglia 插件web页面¶

    # 1.在前面下载解压的目录中，将web模板拷贝到ganglia页面的根目录中
cd ganglia_jobmonarch-1.1.2
cp -a web/templates/job_monarch /var/www/html/templates
mkdir /var/www/html/addons
cp -a web/addons/job_monarch /var/www/html/addons

# 2.配置web页面
vi /var/www/html/conf_default.php
==> $conf['template_name'] = "job_monarch";
vim /var/www/html/addons/job_monarch/conf.php

    <?php
// Show hosts in a jobview by default?
//
$default_showhosts = true;

// Stop displaying archive search results after SEARCH_RESULT_LIMIT
//
$SEARCH_RESULT_LIMIT = 20;

// Show the column job attribute 'requested memory'?
//
$COLUMN_REQUESTED_MEMORY = false;

// Show the column job attribute 'queued' (since)?
//
$COLUMN_QUEUED = true;

// Show the column job attribute 'nodes' hostnames?
//
$COLUMN_NODES = true;

// Path to Ganglia's web frontend root
//
// $GANGLIA_PATH = "/data/ganglia-web-current/";
$GANGLIA_PATH = "/var/www/html";                           `这里写ganglia页面的根目录`
//$GANGLIA_PATH = "../..";

// Format of how to display a date and time in human readable format
//
$DATETIME_FORMAT = "%a %d %b %Y %H:%M:%S";

// Max size of small clusterimage
// (250 pixels is same width as Ganglia's pie chart)
//
$SMALL_CLUSTERIMAGE_MAXWIDTH = 300;

// The size of a single node in the small clusterimage
//
$SMALL_CLUSTERIMAGE_NODEWIDTH = 7;

// Max size of small clusterimage
// (250 pixels is same width as Ganglia's pie chart)
//
$BIG_CLUSTERIMAGE_MAXWIDTH = 500;

// The size of a single node in the small clusterimage
//
$BIG_CLUSTERIMAGE_NODEWIDTH = 11;

// Max size of small host image
//
$SMALL_HOSTIMAGE_MAXWIDTH = 450;

// How to mark nodes with a job in clusterimage
//
$JOB_NODE_MARKING = "J";

// How to mark nodes that are 'down' in the batch
//
$NODE_DOWN_MARKING = "X";

// How to mark nodes that are 'offline' in the batch
//
$NODE_OFFLINE_MARKING = ".";

// Show empty columns in the ClusterImage?
//
$SHOW_EMPTY_COLUMN = false;

// Show empty rows in the ClusterImage?
//
$SHOW_EMPTY_ROW = true;

// XML Datasource for Job Monarch
// by default localhost's gmetad
// [syntax: <ip>:<port>]
//
$DATA_SOURCE = '127.0.0.1:8652';                            `这里写本地地址即可`

// Is there a jobarchive?
//
$JOB_ARCHIVE = true;

// Path to the job archive rrd files
//
$JOB_ARCHIVE_DIR = "/user/local/jobmonarch";                `这里写上一步中设置的rrd数据存放的路径`

// Location of the job archive database
// [syntax: <ip>/<dbase>]
//
$JOB_ARCHIVE_DBASE = "127.0.0.1/jobarchive";                `数据库地址和数据库名称`

$JOB_ARCHIVE_SQL_USER = "postgres";                         `数据库用户名`
$JOB_ARCHIVE_SQL_PASSWORD = "1qaz2WSX";                     `数据库密码`

// Include cluster specific settings here, 
// they will override any (global) settings above
// on a per-cluster basis, where available.
//
//$CLUSTER_CONFS["Example Cluster"] = "./clusterconf/example.php";
$CLUSTER_CONFS["LISA Cluster"]  = "./clusterconf/lisa.php";
//
?>

    1.7 部署配置jobmond¶

    # 1.在slurm服务端下载插件文件并解压
wget http://ftp.surfsara.nl/pub/outgoing/jobmonarch/1.1.2/ganglia_jobmonarch-1.1.2.tar.gz
tar xf ganglia_jobmonarch-1.1.2.tar.gz

# 2.部署jobmond
cd ganglia_jobmonarch-1.1.2
cp -a jobmond/jobmond.py /usr/local/sbin/jobmond.py
cp -a jobmond/jobmond.conf /etc/jobmond.conf

# 3.编辑配置文件
vim /etc/jobmond.conf

    [DEFAULT]
# Specify debugging level here;
#
# 10 = gemtric cmd's
#
DEBUG_LEVEL     : 0

# Wether or not to run as a daemon in background
#
DAEMONIZE       : 1

# What Batch type is the system
# 
# Currently supported: pbs, slurm, sge (experimental), lsf (experimental)
#
BATCH_API       : slurm                                         `监控的调度系统类型`

# Which Batch server to monitor
#
BATCH_SERVER        : localhost                                 `slurm服务端地址`

# Which queue(s) to report jobs of
# (optional)
#
#QUEUE          : long, short

# How many seconds interval for polling of jobs
#
# this will effect directly how accurate the
# end time of a job can be determined
#
BATCH_POLL_INTERVAL : 30

# Location of gmond.conf
#
# Default: /etc/gmond.conf      
                                                                `这里没有设置，会默认寻找/etc/gmond.conf`
#
# DEPRECATED!:      use GMETRIC_TARGET!
#
#GMOND_CONF     : /etc/gmond.conf
#
# Location of gmetric binary
#
# Default: /usr/bin/gmetric
#
# DEPRECATED!:      use GMETRIC_TARGET!
#
#GMETRIC_BINARY     : /usr/bin/gmetric

# Target of Gmetric's: where should we report to
# (usually: your udp_send_channel from gmond)
#
# Syntax: <ip>:<port>
#
#GMETRIC_TARGET     : 172.22.100.30:8649
                                                                `这里不设置，会自动获取gmond.conf中上报给ganglia服务端的地址`
# Enable logging to syslog?
#
USE_SYSLOG                      : 1

# What level msg'es should be logged to syslog?
#
# usually: lvl 0 (errors)
#
SYSLOG_LEVEL                    : 0

# Which facility to use in syslog
#
# Known:
#       KERN, USER, MAIL, DAEMON, AUTH, LPR,
#       NEWS, UUCP, CRON and LOCAL0 through LOCAL7
#
SYSLOG_FACILITY                 : DAEMON


# Wether or not to detect differences in
# time from Torque server and local time.
#
# Ideally both machines (if not the same)
# should have the same time (via ntp or whatever)
#
DETECT_TIME_DIFFS   : 1

# Regexp style hostname translation
#
# Usefull if your Batch hostnames are not the same as your
# Ganglia hostnames (different network interfaces)
#
# Syntax: /orig/new/, /orig/new/
#
BATCH_HOST_TRANSLATE    : 

    # 4.启动作业监控插件
jobmond.py
